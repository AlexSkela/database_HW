CREATE TABLE IF NOT EXISTS Genre (
Id SERIAL PRIMARY KEY,
Name_genre VARCHAR(80) UNIQUE NOT NULL
);

CREATE TABLE IF NOT EXISTS Singer (
Id SERIAL PRIMARY KEY,
Name VARCHAR(80) NOT NULL UNIQUE
);

CREATE TABLE IF NOT EXISTS GenderSinger (
Id_singer INTEGER REFERENCES Singer(Id) NOT NULL,
Id_genre INTEGER REFERENCES Genre(Id) NOT NULL,
CONSTRAINT GenderSinger_pk PRIMARY KEY (Id_singer, Id_genre)
);

CREATE TABLE IF NOT EXISTS Album (
Id SERIAL PRIMARY KEY,
YearOfRelease INTEGER NOT NULL,
NameAlbum VARCHAR(80) NOT NULL
);

CREATE TABLE IF NOT EXISTS AlbumSinger (
Id_singer INTEGER REFERENCES Singer(Id) NOT NULL,
Id_album INTEGER REFERENCES Album(Id) NOT NULL,
CONSTRAINT AlbumSinger_pk PRIMARY KEY (Id_singer, Id_album)
);

CREATE TABLE IF NOT EXISTS Track (
Id SERIAL PRIMARY KEY,
NameOfTrack VARCHAR(50) NOT NULL,
Duration NUMERIC,
Id_album INTEGER REFERENCES Album(Id) NOT NULL
);

CREATE TABLE IF NOT EXISTS Collection (
Id SERIAL PRIMARY KEY,
Name_collection VARCHAR(80) NOT NULL,
YearOfRelease INTEGER NOT NULL
);

CREATE TABLE IF NOT EXISTS CollectionTrack (
Id_collection INTEGER REFERENCES Collection(Id) NOT NULL,
Id_track INTEGER REFERENCES Collection(Id) NOT NULL,
CONSTRAINT CollectionTrack_pk PRIMARY KEY (Id_collection, Id_track)
);
